    // "linkcol" transparent embed which works without a numeric ID:

  },
  {
    title: "Pink Camouflage",
    slug: "pink-camouflage",

    art: "https://f4.bcbits.com/img/a2064831983_7.jpg",
    released: "Feb 2024",
    price: "$1.00 CAD"
  }
];

// NOTE: The numeric track IDs in embedSrc above for Open And Alive and Pink Camouflage
// are placeholders — see instructions in the README comment below.
// The two confirmed IDs (from page source) are:
//   integers-trio-version → 1751975508
//   needle-park           → 2971554013
// For the other two, follow the "Finding your track IDs" note at the bottom of this file.

let activeIdx = null;
const grid = document.getElementById('track-grid');
const playerBar = document.getElementById('player-bar');
const bcPlayer = document.getElementById('bc-player');
const playerTitle = document.getElementById('player-title');
const playerClose = document.getElementById('player-close');

tracks.forEach((track, i) => {
  const card = document.createElement('div');
  card.className = 'track-card';
  card.setAttribute('data-idx', i);
  card.innerHTML = `
    <img class="track-art" src="${track.art}" alt="${track.title}">
    <div class="track-overlay">
      <div class="play-ring">
        <svg class="icon-play" viewBox="0 0 16 16"><polygon points="3,2 14,8 3,14"/></svg>
        <svg class="icon-stop" viewBox="0 0 16 16"><rect x="3" y="2" width="4" height="12" rx="1"/><rect x="9" y="2" width="4" height="12" rx="1"/></svg>
      </div>
    </div>
    <div class="track-info">
      <div class="track-title">${track.title}</div>
      <div class="track-meta">${track.released}</div>
      <div class="track-price">${track.price}</div>
    </div>
  `;
  card.addEventListener('click', () => selectTrack(i));
  grid.appendChild(card);
});

function selectTrack(idx) {
  if (activeIdx === idx) return;

  document.querySelectorAll('.track-card').forEach(c => c.classList.remove('active'));
  document.querySelector(`.track-card[data-idx="${idx}"]`).classList.add('active');

  activeIdx = idx;
  const track = tracks[idx];

  // Swap the src — this naturally stops any previous playback
  bcPlayer.src = '';
  requestAnimationFrame(() => {
    bcPlayer.src = track.embedSrc;
  });

  playerTitle.textContent = track.title;
  playerBar.classList.add('visible');
}

playerClose.addEventListener('click', () => {
  playerBar.classList.remove('visible');
  bcPlayer.src = '';
  document.querySelectorAll('.track-card').forEach(c => c.classList.remove('active'));
  activeIdx = null;
});

     That number is your track ID.
